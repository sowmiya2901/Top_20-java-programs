  public class reverse_array {
	
	public static void main(String[] args) {
		 int arr[]=new int[] {1,4,5};
		 int arr2[]=new int[arr.length];
		 int n=arr2.length;
		
		for(int i=0;i<n;i++) {
			 arr2[n-(i+1)]=arr[i];
		}
		for(int i=0;i<n;i++) {
			System.out.print(arr2[i]);
		}
	}

}
