public class duplicates_array {
	
	public static void main(String[] args) {
	     char a[]=new char[] {'a','s','d','s'};
	     int n=a.length;
	     for(int i=0;i<n;i++) {
	    	 if(a[i]!=0) {
	    	 for( int j=i+1;j<n;j++) {
	    	  if(a[i]==a[j]) {
	    		         a[j]=0;
	    	    		
	    			 }
	    		 }
	    		
	    	 }
	     }
	     for(int i=0;i<n;i++) {
	    	 System.out.println(a[i]); 
	     }
	     }
	    
	}
